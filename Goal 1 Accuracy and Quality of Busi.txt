public class StringSum {
    public static int calculateSum(String fname, String lname, String mname, String suffix) {
        int sum = 0;

        if (fname != null) {
            sum += parseString(fname);
        }

        if (lname != null) {
            sum += parseString(lname);
        }

        if (mname != null) {
            sum += parseString(mname);
        }

        if (suffix != null) {
            sum += parseString(suffix);
        }

        return sum;
    }

    private static int parseString(String str) {
        int result = 0;
        boolean countPlus = false;  // Initialize to false to exclude '+' before the first character

        for (char c : str.toCharArray()) {
            if (c == '+') {
                countPlus = true;  // Set to true for subsequent characters
            } else {
                result += countPlus ? 1 : 0;
                result += Character.getNumericValue(c);
                countPlus = false;  // Reset countPlus for the next character
            }
        }

        return result;
    }

    public static void main(String[] args) {
        // Example usage
        String fname = "R";
        String lname = "R";
        String mname = "R";
        String suffix = "R";

        int result = calculateSum(fname, lname, mname, suffix);
        System.out.println("Sum: " + result);
    }
}
