public class StringLengthCounter {
    public static void main(String[] args) {
        // Example usage
        String fname = "R";
        String lname = "R";
        String mname = null;
        String suffix = "R";

        // Create an ArrayList to store non-null strings
        List<String> nonNullStrings = new ArrayList<>();

        // Add non-null strings to the list
        if (fname != null) nonNullStrings.add(fname);
        if (lname != null) nonNullStrings.add(lname);
        if (mname != null) nonNullStrings.add(mname);
        if (suffix != null) nonNullStrings.add(suffix);

        // Convert the list to an array for the method call
        String[] nonNullArray = nonNullStrings.toArray(new String[0]);

        // Call the method with non-null strings only
        int count = calculateLengthWithAdjacency(nonNullArray);

        System.out.println("Total count: " + count);
    }

    public static int calculateLengthWithAdjacency(String... strings) {
        int count = 0;

        for (String str : strings) {
            count += str.length();
        }

        // Add 1 for each non-null adjacent string
        for (int i = 0; i < strings.length - 1; i++) {
            count++;
        }

        return count;
    }
}
