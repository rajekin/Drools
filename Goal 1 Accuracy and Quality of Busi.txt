public class StringSum {
    public static int calculateSum(String fname, String lname, String mname, String suffix) {
        int sum = 0;

        sum += calculateLength(fname, lname);
        sum += calculateLength(lname, mname);
        sum += calculateLength(mname, suffix);
        sum += calculateLength(suffix, null);  // Since there is no next argument after suffix

        return sum;
    }

    private static int calculateLength(String str, String nextStr) {
        if (str == null || str.isEmpty()) {
            System.out.println("String is null or empty.");
            return 0;
        }

        int length = str.length();

        // Add +1 after count of the first argument if the next argument is not null
        if (nextStr != null && !nextStr.isEmpty() && length > 1) {
            length++;
            System.out.println("Added +1 after the count of the first argument.");
        }

        System.out.println("Length of \"" + str + "\": " + length);
        return length;
    }

    public static void main(String[] args) {
        // Example usage
        String fname = "R";
        String lname = null;
        String mname = null;
        String suffix = "R";

        int result = calculateSum(fname, lname, mname, suffix);
        System.out.println("Total sum of lengths: " + result);
    }
}
