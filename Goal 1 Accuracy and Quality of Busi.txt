import java.util.Arrays;

public class ArrayObjectUtil {

    public static boolean hasHolidaysGreaterThanZero(MyObject[] array) {
        // Check if array is not null and not empty
        if (array != null && array.length > 0) {
            // Iterate over the objects using Java 8 stream
            return Arrays.stream(array)
                  .anyMatch(obj -> obj != null && obj.getHolidays() > 0);
        } else {
            // Array is null or empty
            return false;
        }
    }

    public static void main(String[] args) {
        // Example usage:
        MyObject[] objects = {new MyObject(0), new MyObject(3), new MyObject(-1)};
        boolean result = hasHolidaysGreaterThanZero(objects);
        System.out.println("Any object with holidays greater than 0? " + result);
    }
}
