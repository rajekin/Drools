import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.File;
import java.io.IOException;

public class JSONReader {
    public static void main(String[] args) {
        ObjectMapper mapper = new ObjectMapper();
        File jsonFile = new File("data.json"); // Replace "data.json" with your JSON file path
        
        try {
            // Read JSON from file and map it to a Java object
            YourObject yourObject = mapper.readValue(jsonFile, YourObject.class);
            
            // Now you have your Java object (YourObject) containing the data from the JSON file
            // Handle the Java object as needed
            System.out.println("Object created from JSON: " + yourObject.toString());
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
