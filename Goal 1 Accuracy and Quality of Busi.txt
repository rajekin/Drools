List<C> cList = memberList.stream()
                .map(member -> {
                    C c = new C();
                    c.setName(member.getName());
                    c.setAge(member.getAge());
                    return c;
                })
                .collect(Collectors.toList());

        if (member2List != null && !member2List.isEmpty()) {
            List<C> cListFromSecondList = member2List.stream()
                    .map(member -> {
                        C c = new C();
                        c.setName(member.getName());
                        c.setAge(member.getAge());
                        return c;
                    })
                    .collect(Collectors.toList());

            cList.addAll(cListFromSecondList);
        }

        return cList;
