import com.google.gson.Gson;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class JSONToJavaUsingGson {
    public static void main(String[] args) {
        try {
            // Read JSON from file
            BufferedReader reader = new BufferedReader(new FileReader("data.json")); // Replace "data.json" with your JSON file path
            StringBuilder jsonBuilder = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                jsonBuilder.append(line);
            }
            reader.close();

            String jsonData = jsonBuilder.toString();

            // Convert JSON to Java object using Gson
            Gson gson = new Gson();
            YourObject yourObject = gson.fromJson(jsonData, YourObject.class);

            // Now you have your Java object (YourObject) containing the data from the JSON file
            // Handle the Java object as needed
            System.out.println("Object created from JSON: " + yourObject.toString());
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
