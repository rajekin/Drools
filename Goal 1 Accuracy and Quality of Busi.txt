import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;

public class StylingNameSorter {

    public static ArrayList<StylingName> sortByNameLineLengthDescending(ArrayList<StylingName> stylingNames) {
        // Sort the ArrayList using a custom Comparator
        Collections.sort(stylingNames, new Comparator<StylingName>() {
            @Override
            public int compare(StylingName s1, StylingName s2) {
                // Compare the length of nameLine in descending order
                return Integer.compare(s2.getNameLine().length(), s1.getNameLine().length());
            }
        });

        return stylingNames;
    }

    public static void main(String[] args) {
        // Example usage
        ArrayList<StylingName> stylingNames = new ArrayList<>();
        stylingNames.add(new StylingName("Name 1"));
        stylingNames.add(new StylingName("Longer Name 2"));
        stylingNames.add(new StylingName("Name 3"));

        ArrayList<StylingName> sortedList = sortByNameLineLengthDescending(stylingNames);

        // Print the sorted list
        for (StylingName sn : sortedList) {
            System.out.println(sn.getNameLine());
        }
    }
}

class StylingName {
    private String nameLine;

    public StylingName(String nameLine) {
        this.nameLine = nameLine;
    }

    public String getNameLine() {
        return nameLine;
    }
}
